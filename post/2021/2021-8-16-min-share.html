export default function(params, cb) {
  useShareAppMessage((props) => {
    if (props.from === "button") {
    }
    let _path = `${urlStringify(
      params.path || "/pages/index/index",
      params.query || {}
    )}${Object.keys(params.query).length ? "&" : "?"}cb=${Aes.encrypt(
      params.cb
    )}`;
    cb && cb(props,params);

      title: params.title,
      path: _path,
      imageUrl: params.imageUrl,
    };
  });
  return;
